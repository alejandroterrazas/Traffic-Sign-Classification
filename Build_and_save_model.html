The purpose of this notebook is to build and save a convolutional neural net model.

  • The model is trained to recognize German street signs.

  • The notebook also has a data augmentation scheme for generating variants of the ground truth images.

  • Once training is complete, it is saved for later testing and visualiation.

Data augmentation is necessary to generate enough training examples and, equally importantly, to have balanced data.

The dataset is unbalanced and, therefore, underrepresented classes need to be boosted to the level of overrepresented classes. Moreover, the overrepresented classes are not sufficiently large. In order to generate batches for train, validation, and test, an iterator was created. Using this iterator, a batch can be requested "on demand". The batch is a mix of the original data provided in the pickle file plus synthetic examples sufficient to crate a balanced (i.e., each class is equally represented) batch of images. Note: Due to time constraints, shrinking of the image was not used but would improvie performance of the synthetic image generation.

def gen_data(image):
   imdim = np.shape(image)[0]
   ##generate random size; allow for 25% of image wideth
   newsize = random.randint(imdim,imdim+imdim//4)
   
   image_resized = resize(image, (newsize, newsize, 3),
                          mode='constant',cval=0, 
                          preserve_range=True, clip=True)
   #if newsize < imdim:

   center = (imdim//2, imdim//2)
   M = cv2.getRotationMatrix2D(center, random.randint(-12,12), 1.0) 
   rotated = cv2.warpAffine(image_resized, M, (imdim, imdim))
   ##edgesize tells us max offset for translation
   edgesize = (newsize-imdim)//2

   noisy = random_noise(image_resized, var=sigma*2)

def dense_to_one_hot(labels_dense, num_classes):
  """Convert class labels from scalars to one-hot vectors."""
  num_labels = labels_dense.shape[0]
  index_offset = np.arange(num_labels) * num_classes
  labels_one_hot = np.zeros((num_labels, num_classes))
  labels_one_hot.flat[index_offset + labels_dense.ravel()] = 1
  return labels_one_hot

         image_number = label_indices[random.randint(0,n_labels-1)]
       
         lab = cv2.cvtColor(X_data[image_number,:,:,:], cv2.COLOR_BGR2LAB)  # convert from BGR to LAB color space
         l, a, b = cv2.split(lab)  # split on 3 different channels

         l2 = clahe.apply(l)  # apply CLAHE to the L-channel
 
         lab = cv2.merge((l2,a,b))  # merge channels
         source_image = cv2.cvtColor(lab, cv2.COLOR_LAB2BGR).astype('float')          
         #source_image = X_data[image_number,:,:,:].astype('float')
  
         shape = source_image.shape
         ##ravel flattens the image
         source_image = minmax_scale(source_image.ravel(), 
                        feature_range=(0,1)).reshape(shape)

         ##decide whether to use the source image or generate a synthetic
         ##using randint.  The more source images you have the less likely
         ##you are to generate an image.  n_samples is the total number of
         ##samples/class desired.  
         #
         if random.randint(0,n_samples) > n_labels:
            X[indx,:,:,:] = gen_data(source_image)
         else:
            X[indx,:,:,:] = source_image
            
         y_one_hot = dense_to_one_hot(np.array(labels), n_classes)
        
      yield X, y_one_hot

import tensorflow as tf

initializer = tf.contrib.layers.xavier_initializer()

weights = {
    'wc1': tf.Variable(initializer([3, 3, 3, 32])),
    'wc2': tf.Variable(initializer([3, 3, 32, 256])),
    'wc3': tf.Variable(initializer([3, 3, 256, 512 ])),   
    'wc4': tf.Variable(initializer([3, 3, 512, 1024])),
    'wc5': tf.Variable(initializer([1, 1, 1024, 64 ])),
    'wd1': tf.Variable(initializer([2*2*64, 512])),
    'wd2': tf.Variable(initializer([512,1024])),
    'out': tf.Variable(initializer([1024, n_classes]))}

biases = {
    'bc1': tf.Variable(initializer([32])),
    'bc2': tf.Variable(initializer([256])),
    'bc3': tf.Variable(initializer([512])),
    'bc4': tf.Variable(initializer([1024])),
    'bc5': tf.Variable(initializer([64])),
    'bd1': tf.Variable(initializer([512])),
    'bd2': tf.Variable(initializer([1024])),    
    'out': tf.Variable(initializer([n_classes]))}

def conv2d(x, W, b, strides=1):
    x = tf.nn.conv2d(x, W, strides=[1, strides, strides, 1], padding='SAME')
    x = tf.nn.bias_add(x, b)
    return tf.contrib.layers.batch_norm(tf.nn.relu(x))

def maxpool2d(x, k=2):
    return tf.nn.max_pool(x, ksize=[1, k, k, 1], strides=[1, k, k, 1], padding='SAME')


def conv_net(x, weights, biases, dropout):
    # Layer 1 - 32*32*3 to 16*16*32
    conv1 = conv2d(x, weights['wc1'], biases['bc1'], strides=1)
    conv1 = maxpool2d(conv1, k=2)
    conv1 = tf.nn.lrn(conv1, 4, 
                      bias=1.0, alpha=0.001 / 5.0, 
                      beta=0.5, name="conv1")      
    
    # Layer 2 - 16*16*32 to 16*16*256
    conv2 = conv2d(conv1, weights['wc2'], biases['bc2'], strides=1)
    conv2 = maxpool2d(conv2, k=1)
    conv2 = tf.nn.lrn(conv2, 4, 
                      bias=1.0, alpha=0.001 / 5.0, 
                      beta=0.5, name='conv2') 
    
    # Layer 3 - 16x16*128 to 8x8*512
    conv3 = conv2d(conv2, weights['wc3'], biases['bc3'], strides=2)
    conv3 = maxpool2d(conv3, k=1)
    conv3 = tf.nn.lrn(conv3, 4, 
                      bias=1.0, alpha=0.001 / 5.0, 
                      beta=0.5, name='conv3') 
    
    # Layer 4 - 8x8*512 to 2x2*1024    
    conv4 = conv2d(conv3, weights['wc4'], biases['bc4'], strides=2)
    conv4 = maxpool2d(conv4, k=2)
    conv4 = tf.nn.lrn(conv4, 4, 
                      bias=1.0, alpha=0.001 / 5.0, 
                      beta=0.5, name='conv4') 
    
    # Layer 5 - compress feature maps with a 1x1 to 64
    conv5 = conv2d(conv4, weights['wc5'], biases['bc5'], strides=1)
    conv5 = maxpool2d(conv5, k=1)
    conv5 = tf.nn.lrn(conv5, 4, 
                      bias=1.0, alpha=0.001 / 5.0, 
                      beta=0.5, name='conv5') 
    
    # Fully connected layer 1- 2*2*64 to 1024
    fc1 = tf.reshape(conv5, [-1, weights['wd1'].get_shape().as_list()[0]])
    fc1 = tf.add(tf.matmul(fc1, weights['wd1']), biases['bd1'])
    fc1 = tf.nn.relu(fc1)
    fc1 = tf.nn.dropout(fc1, dropout)
 
    # Fully connected layer 1- 1024 to 512   
    fc2 = tf.add(tf.matmul(fc1, weights['wd2']), biases['bd2'])
    fc2 = tf.nn.relu(fc2)
    fc2 = tf.nn.dropout(fc2, dropout)
    
    # Output Layer - class prediction - 512 to 43
    out = tf.add(tf.matmul(fc2, weights['out']), biases['out'])
    
    return out

run_number = 0

#Launch the session to run the graph over training epochs
with tf.Session() as sess:
    sess.run(init)
    
    for epoch in range(epochs):
        #make the batch_size progressively bigger for each epoch
        #Note: this helps with continuing the convergence
        batch_size += epoch*64
        print("***EPOCH: ", epoch, "  BATCH SIZE: ", batch_size)
        
        for batch in range(n_train//batch_size):
            run_number += 1
            ##grab another batch for training
            batch_x, batch_y = next(train_iterator)
            
            sess.run(optimizer, feed_dict={x: batch_x, 
                                           y: batch_y, 
                                           keep_prob: 0.5}) 
           
            loss = sess.run(cost, feed_dict={x: batch_x, 
                                             y: batch_y, 
                                             keep_prob: 1.0})
            
            loss_vals.append(loss)
        

            batch_x, batch_y = next(valid_iterator)      
            valid_acc = sess.run(accuracy, 
                                 feed_dict={x: 
                                            batch_x, 
                                            y: batch_y,
                                            keep_prob: 1.})
            
            valid_accuracy.append(valid_acc)
  
 
            batch_x, batch_y = next(train_iterator)
        
            train_acc = sess.run(accuracy, 
                                 feed_dict={x: batch_x, 
                                            y: batch_y,
                                            keep_prob: 1.})
            
            train_accuracy.append(train_acc)
            
            if run_number%10 == 0:
              print('Epoch {:>2}, Batch {:>3} - Loss: {:>10.4f} Validation Accuracy: {:.6f}'.

    saver.save(sess, 'model_final')
