Run the model on street signs from the Internet¶

  • Ten images were downloaded and cropped to 32x32x3
  • The program outputs an example of the most likely class on one row (large) plus the 4 other candidates from the top 5 in order of likelihood.

import glob
import tensorflow as tf
import cv2
import numpy as np
from matplotlib import pyplot as plt

testimages = glob.glob("./cropped*")
print(testimages)

tf.reset_default_graph()

graph = tf.get_default_graph()
#init = tf.global_variables_initializer()
saver = tf.train.import_meta_graph('./model_final.meta') 

with tf.Session() as sess:

    saver.restore(sess,tf.train.latest_checkpoint('./'))
    cpred = graph.get_tensor_by_name("correct_pred:0")
    pvec = graph.get_tensor_by_name("prob_vec:0")
    
 
 
    accuracy = graph.get_tensor_by_name("accuracy:0")    
    #print(accuracy)
    for imagename in testimages:
       testimage = cv2.cvtColor(cv2.imread(imagename), 
                                cv2.COLOR_BGR2RGB)
       scaled_test = testimage/np.max(testimage)

       plt.imshow(testimage)
         
       print("****TEST IMAGE:", imagename, "******")
  
       plt.title("TEST IMAGE")
       plt.show()
    
       testvec = np.zeros([1,32,32,3])
       testvec[0,:,:,:] = testimage
       
       pred = sess.run(pvec, 
                       feed_dict={"x:0": testvec,
                                  "keep_prob:0": 1.})
       #print(pred)
       sorted_indices = np.argsort(pred)[0]
       s_i = sorted_indices[::-1][:5]
       
       y_index = np.where(y_test==s_i[0])
       plt.imshow(X_test[y_index[0][0],:,:,:])
       plt.title("Top Match")
       plt.show()
       for i in range(1,5):
          plt.subplot(1,4,i)
          y_index = np.where(y_test==s_i[i])            
          plt.imshow(X_test[y_index[0][0],:,:,:])
          plt.title("Match # {}".format(i+1))
       plt.show()
       print("top 5 probs", pred[0][s_i])
       print("top_5_index", s_i)
       print("************")
